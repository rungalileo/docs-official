```python Python
from galileo import GalileoLogger
logger = GalileoLogger()

## Create a workflow span (parent span)
workflowSpan = logger.addWorkflowSpan(
    input="Processing workflow",
    name="Main Workflow",
    metadata={ workflow_type: "test" },
    tags=["workflow"])
```
